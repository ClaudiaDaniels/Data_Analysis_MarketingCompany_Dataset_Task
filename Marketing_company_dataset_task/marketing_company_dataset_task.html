Scatter Plot: scatter plot of Clicks vs Impressions

In[98]:
Impressions = [13116.0, 25470.0, 403440.0, 50573.0, 56573.0, 65120.0, 95747.0]  # Impressions
Clicks = [490.0, 1593.0, 2059.0, 2585.0, 3100.0, 4510.0, 4948.0]  # Clicks
plat.scatter(Impressions, Clicks, color='green', label='Data Points')
z = nup.polyfit(Impressions, Clicks, 1)  # 1 indicates a linear fit (degree 1)
p = nup.poly1d(z)
plat.plot(Impressions, p(Impressions), color='red', linestyle='--', label='Trendline')
plat.xlabel('Impressions')
plat.ylabel('Clicks')
plat.title('Scatter Plot of Clicks vs Impressions with Trendline')
plat.legend()
plat.show()
